    # vue-cli3 笔记

    # 版本:3.11.0

    vue-cli3 总体是采用 lerna 来管理所有包的，主要的核心代码在packages/@vue组织下

slash:转换 Windows 反斜杠转换为正斜杠

commander: 基于nodejs的命令行处理工具
envinfo: 获取开发环境的系统，包，工具等信息

    # vue upgrade

    vue upgrade [plugin-name] 主要用来升级 cli-service 和 plugins,可以指定对应版本

    program
  .command('upgrade [plugin-name]')
  .description('(experimental) upgrade vue cli service / plugins')
  // 指定对应的版本
  .option(
    '-t, --to <version>',
    'upgrade <package-name> to a version that is not latest',
  )
  // 定义npm仓库
  .option(
    '-r, --registry <url>',
    'Use specified npm registry when installing dependencies',
  )
  // 更新所有插件
  .option('--all', 'Upgrade all plugins')
  // 更新至alpha/beta/rc等版本
  .option('--next', 'Also check for alpha / beta / rc versions when upgrading')
  .action((packageName, cmd) => {
    // 加载upgrade文件，传递包的名称和参数
    require('../lib/upgrade')(packageName, cleanArgs(cmd));
  });

    当我们执行 vue upgrade 的时候，会把包的名称和参数传入到lib/upgrade所暴露出来的函数中去。

    # upgrade 文件主要暴露了一个 upgrade 函数
    /* lib/upgrade.js */
async function upgrade(packageName, options, context = process.cwd()) {
  if (!(await confirmIfGitDirty(context))) {
    return;
  }

  const upgrader = new Upgrader(context);

  if (!packageName) {
    if (options.to) {
      error(`Must specify a package name to upgrade to ${options.to}`);
      process.exit(1);
    }

    if (options.all) {
      return upgrader.upgradeAll(options.next);
    }

    const upgradable = await upgrader.checkForUpdates(options.next);
    if (upgradable) {
      const {ok} = await inquirer.prompt([
        {
          name: 'ok',
          type: 'confirm',
          message: 'Continue to upgrade these plugins?',
          default: true,
        },
      ]);

      if (ok) {
        return upgrader.upgradeAll(options.next);
      }
    }

    return;
  }

  return upgrader.upgrade(packageName, options);
}

          npmGlobalPackages: ['@vue/cli'],
        },
        {
          showNotFound: true,
          duplicates: true,
          fullTree: true,
        },
      )
      .then(console.log);
  });
